Paso 2: Crear procedimiento de relleno de la tabla
Creamos el procedimiento insertamuestras que reciba como par√°metro la cantidad de registros a insertar y los introduzca en la tabla con valores generados aleatoriamente. Para cada muestra generada deberemos obtener los siguientes datos:

Dato Descripci√≥n
Ciudad Obtendremos un n√∫mero entero aleatorio de 3 posibles valores (0, 1 y 2) y dependiendo del resultado grabaremos ('Alicante', 'Valencia' o 'Castell√≥n').

Toma Obtendremos un n√∫mero entero aleatorio de 2 posibles valores (0 y 1) y dependiendo del resultado grabaremos ('ordinaria' o 'extraordinaria')
Valida Obtendremos un n√∫mero entero aleatorio de 2 posibles valores (0 y 1) que grabaremos, pero deseamos que la probabilidad de obtener 1 sea del 90% y la de obtener 0 del 10%.
Temperatura Obtener una valor decimal entre -10¬∫ y -40¬∫ con una precisi√≥n de 1 decimal. Por ejemplo 15.3¬∫.

Los datos generados para cada muestra constituyen un registro que se tiene que insertar en la tabla.

-- Eliminar el procedimiento si existe
DROP PROCEDURE IF EXISTS insertamuestras;

-- Creaci√≥n del procedimiento
DELIMITER //
CREATE PROCEDURE insertamuestras(pNumReg INT)
BEGIN
    -- DELARACI√ìN DE VARIABLES
    DECLARE contador INT;
    DECLARE numaleatorio INT;
    DECLARE vciudad VARCHAR(50);
    DECLARE vfecha DATE;
    DECLARE vhora TIME;
    DECLARE randhoras INT;
    DECLARE randminutos INT;
    DECLARE randsegundos INT;
    DECLARE vtoma VARCHAR(100);
    DECLARE vvalida tinyint(1);
    DECLARE vtemperatura DECIMAL(5,1);
    DECLARE vhumedad INT;
    DECLARE vviento DECIMAL(6,2);
    
    SET contador := 0;
    WHILE (contador < pNumReg) DO
        -- CIUDAD
        SET numaleatorio := FLOOR(3 * RAND());
        CASE numaleatorio
            WHEN 0 THEN
            SET vciudad := 'Alicante';
            WHEN 1 THEN
            SET vciudad := 'Valencia';
            WHEN 2 THEN
            SET vciudad := 'Castell√≥n';
        END CASE;
        -- FECHA
        SET numaleatorio := FLOOR(1000 * RAND());
        SET vfecha := ADDDATE(CURDATE(), -numaleatorio);
        -- HORA
        SET randhoras := FLOOR(24 * RAND());
        SET randminutos := FLOOR(60 * RAND());
        SET randsegundos := FLOOR(60 * RAND());
        SET vhora := MAKETIME(randhoras, randminutos, randsegundos);
        -- TOMA
        SET numaleatorio := FLOOR(2 * RAND());
        IF (numaleatorio = 0) THEN
            SET vtoma := 'ordinaria';
        ELSE
            SET vtoma := 'extraordinaria';
        END IF;
        -- VALIDA con probabilidad del 90%
        SET numaleatorio := FLOOR(10 * RAND());
        IF (numaleatorio = 0) THEN
            SET vvalida := 0;
        ELSE
            SET vvalida := 1;
        END IF;
        -- TEMPERATURA ENTRE -10¬∫C y 40¬∫C
        SET numaleatorio := FLOOR(501 * RAND()) + (-100);
        SET vtemperatura := numaleatorio / 10;
        -- HUMEDAD en porcentaje de 0% a 99%
        SET numaleatorio := FLOOR(100 * RAND());
        SET vhumedad := numaleatorio;
        -- VIENTO ENTRE 0 Km/h y 149.99 Km/h con dos decimales
        SET numaleatorio := FLOOR(15000 * RAND());
        SET vviento := numaleatorio / 100;
        -- Insertar registro(id_reg y fecha_act toman valor autom√°ticamente)
        INSERT INTO muestras (ciudad, fecha, hora, toma, valida, temperatura, humedad, viento)
        VALUES (vciudad, vfecha, vhora, vtoma, vvalida, vtemperatura, vhumedad, vviento);
        -- Continuar con el bucle
        SET contador := contador + 1;
    END WHILE;
END //
DELIMITER ;
